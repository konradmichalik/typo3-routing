# cat=basic/enable/010; type=string; label=Path prefix gate:Only request paths starting with one of these prefixes (after stripping the site/language base) are matched against attribute routes. Comma-separated list to support multiple prefixes (e.g. /api/, /va/). This is a pure performance gate; route paths in the #[Route] attribute are still written in full (including the prefix). Leave empty to match every path (with a performance penalty for every page request).
prefix = /api/

# cat=authentication/enable/010; type=string; label=Bearer token env variable name:Default name of the process environment variable that holds the expected bearer token for BearerTokenAuthenticator. Only the variable NAME is stored here, never the secret value. Overridable per route via #[Authenticate(BearerTokenAuthenticator::class, options: ['envName' => '...'])]. Defaults to ROUTING_BEARER_TOKEN when left empty.
bearerTokenEnvName = ROUTING_BEARER_TOKEN

# cat=cors/enable/010; type=string; label=CORS allowed origins:Comma-separated list of origins allowed to call the API (e.g. https://app.example.com, https://admin.example.com). Use * to allow any origin. Leave EMPTY to disable CORS entirely (default). Applies to every matched attribute route; preflight OPTIONS requests are answered automatically.
cors.allowedOrigins =

# cat=cors/enable/040; type=string; label=CORS exposed response headers:Comma-separated list of response headers exposed to the browser (Access-Control-Expose-Headers). Empty by default.
